# Legal Document Clause Extractor using NLP
import spacy
from transformers import pipeline

nlp = spacy.load("en_core_web_sm")
summarizer = pipeline("summarization", model="facebook/bart-large-cnn")

def extract_clauses(document_text: str) -> dict:
    doc = nlp(document_text)
    clauses = {
        "parties": [],
        "obligations": [],
        "dates": [],
        "penalties": []
    }
    for ent in doc.ents:
        if ent.label_ == "PERSON" or ent.label_ == "ORG":
            clauses["parties"].append(ent.text)
        elif ent.label_ == "DATE":
            clauses["dates"].append(ent.text)

    clauses["summary"] = summarizer(
        document_text[:1024],
        max_length=150,
        min_length=40,
        do_sample=False
    )[0]['summary_text']

    return clauses
          

# Legal Document Drafting API (Flask)
from flask import Flask, request, jsonify
from transformers import GPT2LMHeadModel, GPT2Tokenizer

app = Flask(__name__)
tokenizer = GPT2Tokenizer.from_pretrained("gpt2")
model = GPT2LMHeadModel.from_pretrained("gpt2")

@app.route("/draft", methods=["POST"])
def draft_document():
    data = request.json
    doc_type = data.get("type", "NDA")
    party_a = data.get("party_a", "Party A")
    party_b = data.get("party_b", "Party B")

    prompt = (
        f"Draft a professional {doc_type} agreement between "
        f"{party_a} and {party_b}. Terms and conditions:
"
    )
    inputs = tokenizer.encode(prompt, return_tensors="pt")
    outputs = model.generate(inputs, max_length=400, temperature=0.7)
    draft = tokenizer.decode(outputs[0], skip_special_tokens=True)

    return jsonify({"document_type": doc_type, "draft": draft})

if __name__ == "__main__":
    app.run(debug=True)

# YOLOv8 Pothole Detection — Real-Time Inference
from ultralytics import YOLO
import cv2

model = YOLO("pothole_yolov8.pt")  # Fine-tuned YOLOv8 weights

def detect_potholes(video_source=0):
    cap = cv2.VideoCapture(video_source)

    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break

        results = model(frame, conf=0.45)

        for result in results:
            boxes = result.boxes
            for box in boxes:
                x1, y1, x2, y2 = map(int, box.xyxy[0])
                conf = float(box.conf[0])
                label = f"Pothole {conf:.2f}"
                cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 0, 255), 2)
                cv2.putText(frame, label, (x1, y1 - 10),
                            cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 0, 255), 2)

        cv2.imshow("Pothole Detection", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

# NDVI Satellite Analysis for Carbon Credit Validation
import numpy as np
import rasterio

def calculate_ndvi(red_band_path, nir_band_path):
    with rasterio.open(red_band_path) as red_src:
        red = red_src.read(1).astype(float)
    with rasterio.open(nir_band_path) as nir_src:
        nir = nir_src.read(1).astype(float)

    ndvi = (nir - red) / (nir + red + 1e-10)
    carbon_score = np.mean(ndvi[ndvi > 0.3])
    return round(float(carbon_score), 4)
                    

# Carbon Credit Fraud Detection Model
from sklearn.ensemble import IsolationForest
import pandas as pd

def detect_fraud(credit_data: pd.DataFrame):
    features = credit_data[['credit_amount', 'ndvi_score',
                             'land_area', 'previous_claims']]
    model = IsolationForest(contamination=0.05, random_state=42)
    predictions = model.fit_predict(features)
    credit_data['is_fraudulent'] = (predictions == -1).astype(int)
    return credit_data

# Job Relevance Analyzer using NLP
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

def analyze_job_relevance(resume_text: str, job_description: str) -> float:
    vectorizer = TfidfVectorizer(stop_words='english')
    vectors = vectorizer.fit_transform([resume_text, job_description])
    similarity = cosine_similarity(vectors[0:1], vectors[1:2])
    return round(float(similarity[0][0]) * 100, 2)  # Returns match %
